include(FindPkgConfig)

find_package(Qt5Core REQUIRED)
find_package(Qt5Gui REQUIRED)
pkg_check_modules(Qt5PlatformSupport REQUIRED Qt5PlatformSupport)

set(Qt5PlatformSupport_PRIVATE_INCLUDE_DIRS
    ${Qt5PlatformSupport_INCLUDEDIR}/QtPlatformSupport/${Qt5PlatformSupport_VERSION}
    ${Qt5PlatformSupport_INCLUDEDIR}/QtPlatformSupport/${Qt5PlatformSupport_VERSION}/QtPlatformSupport)

set(PLUGIN mock_qpa)

set(PLUGIN_SRCS
    backingstore.cc
    integration.cc
    platformnativeinterface.cc
    plugin.cc
    screen.cc
    shimapi.cc)

add_library(${PLUGIN} SHARED ${PLUGIN_SRCS})
target_link_libraries(${PLUGIN}
    ${Qt5Core_LIBRARIES}
    ${Qt5Gui_LIBRARIES}
    ${Qt5PlatformSupport_LIBRARIES})
target_compile_definitions(${PLUGIN} PRIVATE -DQT_PLUGIN)
target_include_directories(
    ${PLUGIN} PRIVATE
    ${Qt5Core_PRIVATE_INCLUDE_DIRS}
    ${Qt5Gui_PRIVATE_INCLUDE_DIRS}
    ${Qt5PlatformSupport_INCLUDE_DIRS}
    ${Qt5PlatformSupport_PRIVATE_INCLUDE_DIRS})
set_target_properties(
    ${PLUGIN} PROPERTIES
    AUTOMOC TRUE
    LIBRARY_OUTPUT_DIRECTORY ${CMAKE_CURRENT_BINARY_DIR}/../plugins/platforms)
